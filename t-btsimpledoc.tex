%%%% Definitions %%%%%

\def\doctitle#1{\gdef\@title{#1}}
\def\author#1{\gdef\@author{#1}}
\def\date#1{\gdef\@date{#1}}
\date{\currentdate}  % Default to today unless specified otherwise.

\def\maketitle{%
  \startalignment[center]
    \blank[force,3*medium]
      {\tfc \@title}
    \blank[2*medium]
      {\tfa \@author}
    \blank[2*medium]
      {\tfx \@date}
    \blank[3*medium]
  \stopalignment}

\definestartstop
  [abstract]
  [before={\midaligned{\bf Abstract}
           \startnarrower[1*middle]},
   after={\stopnarrower
          \blank[big]}]
